CREATE TABLE artist (code INT NOT NULL AUTO_INCREMENT, 
                     name VARCHAR(20) NOT NULL,
                     birthPlace CHAR(20) NOT NULL, 
                     age INT(11) NOT NULL, 
                     style CHAR(20) NOT NULL,
                     PRIMARY KEY (code)
                     );
                                  
CREATE TABLE artwork(code INT NOT NULL AUTO_INCREMENT, 
                     title CHAR(20) NOT NULL,
                     a_code INT NOT NULL, 
                     year INT(11) NOT NULL,
                     price FLOAT NOT NULL,
                     type CHAR(20) NOT NULL,
                     PRIMARY KEY (code),
                     FOREIGN KEY (a_code) REFERENCES artist (code)
                     );
                     
                     
CREATE TABLE style(code INT NOT NULL AUTO_INCREMENT,
                      group_name VARCHAR(20),
                      PRIMARY KEY (code)
                      );
                      
                      
                      
CREATE TABLE group_art (a_code INT(11),
                       s_code INT(11),
                       PRIMARY KEY (a_code, s_code),
                       FOREIGN KEY (a_code) REFERENCES artwork (code),
                       FOREIGN KEY (s_code) REFERENCES style(code)
                       );
